import requests
from flask import Flask, request, jsonify

app = Flask(__name__)

# [核心配置] 1. 集群内 HolmesGPT 的 Service 访问地址与端口
HOLMES_URL = os.getenv("HOLMES_URL", "http://cluster.local")

# [核心配置] 2. 替换为你之前在 PrometheusAlertCenter 测试通过的完整飞书接收长地址
# ⚠️ 确保 tpl=holmesfeishu 对应你刚才在网页上保存的模版别名
PALERT_URL = os.getenv("PALERT_URL", "http://cluster.local")

@app.route('/webhook', methods=['POST'])
def handle_alert():
    alert_data = request.json
    alerts = alert_data.get("alerts", [])
    
    if not alerts:
        return jsonify({"status": "no alerts"}), 200
        
    print(f"收到 Prometheus 告警，包含 {len(alerts)} 个警报项目。正在启动 HolmesGPT 深度诊断...")
    
    # 遍历处理每一个告警项（防止多告警同时并发时 AI 漏掉）
    for alert in alerts:
        # 只有在告警激活（firing）状态下才触发 AI 诊断，恢复通知（resolved）直接跳过
        if alert.get("status") == "firing":
            alert_name = alert.get("labels", {}).get("alertname", "未知告警")
            summary = alert.get("annotations", {}).get("summary", "无摘要")
            namespace = alert.get("labels", {}).get("namespace", "default")
            
            # 拼装给 Kimi 大脑的运维命令
            ask_text = f"集群在命名空间 {namespace} 触发了 {alert_name} 告警，摘要是: {summary}。请帮我使用 prometheus 监控和 kubectl 查明原因并给出具体结论。"
            
            try:
                # 叩开 HolmesGPT 聊天大门，指定模型别名为 kimi
                holmes_res = requests.post(HOLMES_URL, json={"ask": ask_text, "model": "kimi"}, timeout=90)
                if holmes_res.status_code == 200:
                    analysis_result = holmes_res.json().get("analysis", "HolmesGPT 未返回有效诊断")
                else:
                    analysis_result = f"HolmesGPT 返回非200状态码: {holmes_res.status_code}, 详情: {holmes_res.text}"
            except Exception as e:
                analysis_result = f"调用 HolmesGPT 诊断链路发生异常: {str(e)}"
            
            # 🚀 胶水层最精妙的一步：把 AI 的诊断报告直接强行注入到告警数据的 annotations.analysis 字段中
            if "annotations" not in alert:
                alert["annotations"] = {}
            alert["annotations"]["analysis"] = analysis_result
            print(f"[{alert_name}] 诊断完成，报告已成功注入上下文。")
            
    # 将加工完成、带有 AI 诊断报告的新告警数据包，单向转发给你现有的 PrometheusAlertCenter
    try:
        response = requests.post(PALERT_URL, json=alert_data, timeout=10)
        print(f"已成功将加工后的数据投递至 PrometheusAlert，返回状态: {response.status_code}")
    except Exception as e:
        print(f"向 PrometheusAlert 转发数据失败: {str(e)}")

    return jsonify({"status": "success"}), 200

if __name__ == '__main__':
    # 监听在容器内/宿主机内的 9000 端口
    app.run(host='0.0.0.0', port=9000)
